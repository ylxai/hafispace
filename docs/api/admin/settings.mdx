## Get Settings

Ambil data profil dan pengaturan vendor.

### Response

<ResponseField name="vendor" type="object">
  Data vendor lengkap termasuk pengaturan notifikasi, DP, tema, dll.
</ResponseField>

<ResponseField name="vendor.namaStudio" type="string">
  Nama studio/vendor
</ResponseField>

<ResponseField name="vendor.email" type="string">
  Email vendor
</ResponseField>

<ResponseField name="vendor.phone" type="string">
  Nomor telepon
</ResponseField>

<ResponseField name="vendor.logoUrl" type="string">
  URL logo
</ResponseField>

<ResponseField name="vendor.dpPercentage" type="number">
  Persentase DP (default: 30)
</ResponseField>

<ResponseField name="vendor.rekeningPembayaran" type="string">
  Info rekening pembayaran
</ResponseField>

<ResponseField name="vendor.themeColor" type="string">
  Warna tema (hex)
</ResponseField>

<ResponseField name="vendor.bookingFormActive" type="boolean">
  Status form booking
</ResponseField>

<ResponseField name="vendor.successMessage" type="string">
  Pesan sukses setelah booking
</ResponseField>

<ResponseField name="vendor.syaratKetentuan" type="string">
  Syarat dan ketentuan
</ResponseField>
